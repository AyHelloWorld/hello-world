# 远程分支重命名，先在本地重命名，再删除远程分支，再把本地分支推送远程
gswewf@gswewf-pc:~/yhb$ git branch -m wmd wmd_match
gswewf@gswewf-pc:~/yhb$ git push origin :wmd
gswewf@gswewf-pc:~/yhb$ git push origin wmd_match:wmd_match
